from pyspark.sql.functions import from_json, col, split, trim
from pyspark.sql.types import StructType, StructField, StringType, DoubleType, IntegerType

# Schema for top-level personal_detail JSON (note address is a JSON string)
personal_detail_schema = StructType([
    StructField("person_name", StringType(), True),
    StructField("gender", StringType(), True),
    StructField("address", StringType(), True),  # nested JSON string
    StructField("lat", StringType(), True),
    StructField("long", StringType(), True),
    StructField("city_pop", StringType(), True),
    StructField("job", StringType(), True),
    StructField("dob", StringType(), True)
])

# Schema for the nested address JSON string
address_schema = StructType([
    StructField("street", StringType(), True),
    StructField("city", StringType(), True),
    StructField("state", StringType(), True),
    StructField("zip", StringType(), True)
])

if "personal_detail" in df.columns:

           .withColumn("gender", col("personal_detail_struct.gender")) \

from sklearn.model_selection import train_test_split
